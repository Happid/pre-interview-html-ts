Asynchronous

1. Callback

        
            function getData(myCallback) {
                console.log("Mengambil data...");
                setTimeout(() => {
                    myCallback("Data sudah diterima!");
                });
            }

            getData((result) => {
                console.log(result);
            });

2. Promise

        
            function getData() {
                return new Promise((resolve, reject) => {
                    setTimeout(() => {
                        resolve("Promise Data diterima!");
                    });
                });
            }

            getData().then(result => {
                console.log(result);
            });

3. Async / Await

        
            function getData() {
                return new Promise((resolve) => {
                    setTimeout(() => resolve("Async Await Data masuk!"));
                });
            }

            async function run() {
                console.log("Menunggu data...");
                const result = await getData();
                console.log(result);
            }

            run();

4. Microtask Queue vs Macrotask Queue

        
            console.log("Start");

            setTimeout(() => {
                console.log("Macrotask: Timeout");
            }, 0);

            Promise.resolve()
                .then(() => console.log("Microtask 1"))
                .then(() => console.log("Microtask 2"));

            console.log("End");
