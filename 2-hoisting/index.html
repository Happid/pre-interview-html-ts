Hoisting adalah mekanisme di JavaScript di mana deklarasi variabel dan fungsi “diangkat” ke atas scope sebelum kode dijalankan.

1. var — di-hoist + di-inisialisasi (undefined)

      
        console.log(a); // undefined
        var a = 10;

2. let dan const — di-hoist tapi masuk "Temporal Dead Zone" (TDZ)

      
        console.log(b);  // ❌ ReferenceError
        let b = 10;

3. Function Declaration di-hoist dengan utuh

      
        sayHello(); // Output: Hello!

        function sayHello() {
            console.log("Hello!");
        }

4. Function Expression TIDAK di-hoist

      
        sayHi(); // ❌ TypeError: sayHi is not a function

        var sayHi = function() {
            console.log("Hi!");
        };

5. Hoisting dengan let dalam block scope

      
        {
            console.log(e);  // ❌ ReferenceError (TDZ)
            let e = 50;
        }

6. Hoisting pada class

      
        const user = new Person(); // ❌ ReferenceError: Cannot access 'Person' before initialization

        class Person {
            constructor() {
                this.name = "John";
            }
        }

7. Hoisting pada multiple var declarations

      
        console.log(g); // undefined

        var g = 1;
        var g = 2;

        console.log(g); // 2

8. Hoisting dalam function scope

      
        function test() {
            console.log(a); // undefined
            var a = 100;
        }
        test();

9. TDZ example yang lebih kompleks

      
        let a = 10;

        function demo() {
            console.log(a); // ❌ ReferenceError (karena ada "let a" di scope ini)
            let a = 20;
        }

        demo();
